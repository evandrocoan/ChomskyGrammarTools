36 class LockableType(object):
37  """
38  An object type which can have its attributes changes locked/blocked after its `lock()`
39  method being called.
40 
41  After locking, its string representation attribute is going to be saved as an attribute and
42  returned when needed.
43  """
44 
45  _USE_STRING = True
46  _EMQUOTE_STRING = False
47 
48  def __init__(self):
49  """
50  How to handle call to __setattr__ from __init__?
51  https://stackoverflow.com/questions/3870982/how-to-handle-call-to-setattr-from-init
52  """
53  super().__setattr__('locked', False)

56  self.locked = False

59  self.str = ""

68  def __setattr__(self, name, value):
69  """
70  Block attributes from being changed after it is activated.
71  https://stackoverflow.com/questions/17020115/how-to-use-setattr-correctly-avoiding-infinite-recursion
72  """
73 
74  if self.locked:
75  raise AttributeError( "Attributes cannot be changed after `locked` is set to True! %s" % self.__repr__() )
76 
77  else:
78  super().__setattr__( name, value )

134  def _unlock(self):
135  """
136  Unblock the object changes allowing its attributes to be freely set.
137 
138  Do not call it if this object is inside a hashtable as list or set, because an object
139  cannot have its hash changed while they are inside it. Otherwise, the hashmap will not
140  get updated and the new version of the object will point as non existent.
141  """
142 
143  if not self.locked:
144  return
145 
146  self.__dict__['locked'] = False
147  self._len = self._original_len
148  self._str = self._original_str
149  self._hash = self._original_hash
150 
151  def lock(self):
152  """
153  Block further changes to this object attributes and cache its length and string
154  representation for faster access.
155  """
156 
157  if self.locked:
158  return
159 
160  self.str = str( self )
161  self._str = lambda: self.str
162 
163 
164  self.len = len( self )
165  self._len = lambda: self.len
166 
167  self._hash = hash( self._str() )
168  self.locked = True
169 
170  def new(self, unlocked=True):
171  """
172  Creates and return a new copy of the current a object.
173  """
174  new_copy = copy.deepcopy( self )
175  # log( 1, "self: %s", self )
176  # log( 1, "new_copy: %s", new_copy )
177 
178  if unlocked:
179  new_copy._unlock()
180 
181  else:
182  new_copy.lock()
183 
184  return new_copy
