90  def _parse_symbols(self, symbols, Type):
91  log( 4, 'productions: %s, type: %s', symbols, Type )
92  results = []
93 
94  for symbol in symbols:
95  results.append( str( symbol ) )
96 
97  symbol = Type( "".join( results ) )
98  log( 4, "results: %s %s", results, symbol.repr() )
99  return symbol

101  def production(self, tokens):
102  """
103  Converts the tree leaf Terminal's and NonTerminal's symbols into a production ready to
104  be used in the Chomsky Grammar.
105  """
106  log( 4, 'tokens: %s', tokens )
107  new_production = Production()
108 
109  for element in tokens:
110 
111  if isinstance( element, Tree ):
112  symbol = element.children[0]
113 
114  if isinstance( symbol, ( Terminal, NonTerminal ) ):
115  new_production.add( symbol )
116 
117  log( 4, "new_production: %s", new_production.repr() )
118  return new_production
