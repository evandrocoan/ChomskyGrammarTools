51  """
52  Transforms the AST (Abstract Syntax Tree) nodes into meaningful string representations,
53  allowing simple recursive parsing parsing of the AST tree.
54 
55  Tree(
56  productions,
57  [
58  Tree( non_terminal, [Token( UCASE_LETTER, 'S' )] ),
59  Tree(
60  non_terminals,
61  [
62  Tree( production, [Tree( terminal, [Token( LCASE_LETTER, 'a' )] ), Tree( non_terminal, [Token( UCASE_LETTER, 'A' )] )] ),
63  Tree( production, [Tree( terminal, [Token( LCASE_LETTER, 'a' )] )] )
64  ]
65  )
66  ]
67  )
68  """

267  def _parse_symbol(self, _terminal, default):
268 
269  if len( _terminal ):
270  return Terminal( _terminal )
271 
272  return Terminal( default )

274  def _parse_symbols(self, _symbols, Type):
275  log( 4, 'productions: %s, type: %s', _symbols, Type )
276  results = []
277 
278  for _symbol in _symbols:
279  results.append( str( _symbol ) )
280 
281  symbol = Type( "".join( results ) )
282  log( 4, "results: %s", results )
283  log( 4, "symbol: %s", symbol )
284  return symbol
285 
286  def production(self, productions):
287  """
288  Converts the tree leaf Terminal's and NonTerminal's symbols into a production ready to
289  be used in the Chomsky Grammar.
290  """
291  log( 4, 'productions: %s', productions )
292  new_production = Production()
293 
294  for production in productions:
295 
296  if isinstance( production, ( Terminal, NonTerminal ) ):
297  new_production.add( production )
298 
299  log( 4, "new_production: %s", new_production )
300  return new_production
