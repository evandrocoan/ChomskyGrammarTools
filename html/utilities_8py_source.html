67 def get_duplicated_elements(elements_list):
68  """
69  Given an `elements_list` with duplicated elements, return a set only with the duplicated
70  elements in the list. If there are not duplicated elements, an empty set is returned.
71 
72  How do I find the duplicates in a list and create another list with them?
73  https://stackoverflow.com/questions/9835762/how-do-i-find-the-duplicates-in-a-list-and-create-another-list-with-them
74  """
75  visited_elements = set()
76  visited_and_duplicated = set()
77 
78  add_item_to_visited_elements = visited_elements.add
79  add_item_to_visited_and_duplicated = visited_and_duplicated.add
80 
81  for item in elements_list:
82 
83  if item in visited_elements:
84  add_item_to_visited_and_duplicated(item)
85 
86  else:
87  add_item_to_visited_elements(item)
88 
89  return visited_and_duplicated

92 def trimMessage(message):
93  """
94  Cuts a message maximum width and length and add an ellipsis (...) when it exceeds 30 lines
95  or 150 characters of line size.
96  """
97  message_lines = message.split( '\n' )

100  if len( message_lines ) > 30:
101  message_lines = message_lines[:30]
102  message_lines.append( '\n...' )

104  for line in message_lines:

107  line = line[:150] + '...'

114 def emquote_string(string):
115  """
116  Return a string escape into single or double quotes accordingly to its contents.
117  """
118  string = str( string )
119  is_single = "'" in string
120  is_double = '"' in string
121 
122  if is_single and is_double:
123  return '"{}"'.format( string.replace( "'", "\\'" ) )
124 
125  if is_single:
126  return '"{}"'.format( string )
127 
128  return "'{}'".format( string )

142 def sort_alphabetically_and_by_length(iterable):
143  """
144  Give an `iterable`, sort its elements accordingly to the following criteria:
145  1. Sorts normally by alphabetical order
146  2. Sorts by descending length
147

149  https://stackoverflow.com/questions/4659524/how-to-sort-by-length-of-string-followed-by-alphabetical-order
150  """
151  return sorted( sorted( natsorted( iterable, key=lambda item: str( item ).lower() ),
152  key=lambda item: str( item ).istitle() ),
153  key=lambda item: len( str( item ) ) )

156 def sort_correctly(iterable):
157  """
158  Sort the given iterable in the way that humans expect.
159 
160  How to sort alpha numeric set in python
161  https://stackoverflow.com/questions/2669059/how-to-sort-alpha-numeric-set-in-python
162  """
163  convert = lambda text: int( text ) if text.isdigit() else text
164  alphanum_key = lambda key: [convert( characters ) for characters in re.split( '([0-9]+)', str( key ).lower() )]
165  return sorted( sorted( iterable, key=alphanum_key ), key=lambda item: str( item ).istitle() )

182 def dictionary_to_string(dictionary, sort=None):
183  """
184  Given a dictionary with a list for each string key, call `sort_dictionary_lists()` and
185  return a string representation by line of its entries.
186  """
187 
188  if not len( dictionary ):
189  return " No elements found."
190 
191  if sort:
192  iterable = sort( iterable )
193 
194  strings = []
195  elements_strings = []
196 
197  dictionary = sort_dictionary_lists( dictionary )
198  largest_key = get_largest_item_size( dictionary.keys() ) + 1
199 
200  for key, values in dictionary.items():
201  elements_strings.clear()
202 
203  for item in values:
204  elements_strings.append( "{}".format( str( item ) ) )
205 
206  strings.append( "{:>{largest_key}}: {}".format( str( key ), " ".join( elements_strings ),
207  largest_key=largest_key ) )
208 
209  return "\n".join( strings )

214  Given a dictionary with a list for each string key, call `sort_dictionary_lists()` and

304 def getCleanSpaces(inputText, minimumLength=0, lineCutTrigger="", keepSpaceSepators=False):
305  """
306  Removes spaces and comments from the input expression.
307 
308  `minimumLength` of a line to not be ignored
309  `lineCutTrigger` all lines after a line starting with this string will be ignored
310  `keepSpaceSepators` if True, it will keep at a single space between sentences as `S S`, given `S S`
311  """
312 
313  if keepSpaceSepators:
314  removeNewSpaces = ' '.join( inputText.split( ' ' ) )
315  lineCutTriggerNew = ' '.join( lineCutTrigger.split( ' ' ) ).strip( ' ' )
316 
317  else:
318  removeNewSpaces = re.sub( r"\t| ", "", inputText )
319  lineCutTriggerNew = re.sub( r"\t| ", "", lineCutTrigger )
320 
321  # log( 1, "%s", inputText, minimumLength=0 )
322  lines = removeNewSpaces.split( "\n" )
323  clean_lines = []
324 
325  for line in lines:
326 
327  if keepSpaceSepators:
328  line = line.strip( ' ' )
329 
330  if minimumLength:
331 
332  if len( line ) < minimumLength:
333  continue
334 
335  if lineCutTrigger:
336 
337  if line.startswith( lineCutTriggerNew ):
338  break
339 
340  if line.startswith( "#" ):
341  continue
342 
343  clean_lines.append( line )
344 
345  return clean_lines

348 def wrap_text(text, wrap=0, trim_tabs=False, trim_spaces=False, trim_lines=False, indent=""):
349  """
350  1. Remove input text leading common indentation, trailing white spaces
351  2. If `wrap`, wraps big lists on 80 characters.
352  3. If `trim_spaces`, remove leading '+' symbols and if `trim_tabs` replace tabs with 2 spaces.
353  4. If `trim_lines`, remove all new line characters.
354  """
355  clean_lines = []
356 
357  if not isinstance( text, str ):
358  text = str( text )
359 
360  if trim_tabs:
361  text = text.replace( '\t', ' ' )
362 
363  dedent_lines = textwrap.dedent( text ).strip( '\n' )
364 
365  if trim_spaces:
366 
367  for line in dedent_lines.split( '\n' ):
368  line = line.rstrip( ' ' ).lstrip( '+' )
369  clean_lines.append( line )
370 
371  dedent_lines = textwrap.dedent( "\n".join( clean_lines ) )
372 
373  if wrap:
374  clean_lines.clear()
375 
376  for line in dedent_lines.split( '\n' ):
377  line = textwrap.fill( line, width=wrap, subsequent_indent=indent )
378  clean_lines.append( line )
379 
380  dedent_lines = "\n".join( clean_lines )
381 
382  if trim_lines:
383  dedent_lines = "".join( dedent_lines.split( '\n' ) )
384 
385  return dedent_lines

388 def get_representation(self, ignore=[], emquote=False):
389  """
390  Given a object, iterating through all its public attributes and return then as a string
391  representation.
392 
393  `ignore` a list of attributes to be ignored
394  `emquote` if True, puts the attributes values inside single or double quotes accordingly.
395  """
396  valid_attributes = self.__dict__.keys()
397  clean_attributes = []
398 
399  if emquote:
400 
401  def pack_attribute(string):
402  return emquote_string( string )
403 
404  else:
405 
406  def pack_attribute(string):
407  return string
408 
409  for attribute in valid_attributes:
410 
411  if not attribute.startswith( '_' ) and attribute not in ignore:
412  clean_attributes.append( "{}: {}".format( attribute, pack_attribute( self.__dict__[attribute] ) ) )
413 
414  return "%s %s;" % ( self.__class__.__name__, ", ".join( clean_attributes ) )
