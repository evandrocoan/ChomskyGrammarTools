159  def combinations(self, non_terminal_epsilon):
160  """
161  Return a new set within all its non terminal's removal combinations, accordingly with
162  the non terminal's set on `non_terminal_epsilon`.
163 
164  How do use itertools in Python to build permutation or combination
165  http://thomas-cokelaer.info/blog/2012/11/how-do-use-itertools-in-python-to-build-permutation-or-combination/
166  """
167  # log( 1, "self: %s", self )
168  combinations = set()
169 
170  non_terminals = self.non_terminals( position=True )
171  non_terminals_count = len( non_terminals )
172 
173  for permutation_size in range( 0, non_terminals_count ):
174  n_items_permutation = list( itertools.permutations( non_terminals, permutation_size ) )
175  # log( 1, "permutation_size: %s, n_items_permutation: %s", permutation_size, n_items_permutation )
176 
177  for permutation in n_items_permutation:
178  new_production = self.new()
179 
180  try:
181  new_production.filter_non_terminals( non_terminal_epsilon, permutation )
182 
183  except RuntimeError as error:
184 
185  if str( error ).startswith( "Invalid production creation! Production with no length: [&]" ):
186  new_production = epsilon_production.new()
187 
188  else:
189  raise
190 
191  # log( 1, "new_production: %s (%s)", new_production, repr( new_production ) )
192  combinations.add( new_production )
193 
194  # log( 1, "combinations: \n%s", combinations )
195  return combinations
