84  def __next__(self):
85  """
86  Called by Python automatically when iterating over this set and python wants to know the
87  next element to iterate. Raises `StopIteration` when the iteration has been finished.
88 
89  How to make a custom object iterable?
90  https://stackoverflow.com/questions/21665485/how-to-make-a-custom-object-iterable
91  https://stackoverflow.com/questions/4019971/how-to-implement-iter-self-for-a-container-object-python
92  """
93  empty_slots = self.empty_slots
94  current_index = self.current_index + 1
95  filled_slots = self.filled_slots
96 
97  while current_index in empty_slots or current_index in filled_slots:
98  current_index += 1
99 
100  try:
101  self.current_index = current_index
102  # log( 1, "current_index: %s", current_index )
103  return self.iterable_access( current_index )
104 
105  except IndexError:
106  raise StopIteration

258  def __setitem__(self, key, value):
259  """
260  Given a `key` and `item` add it to this dictionary as a non yet iterated item, replacing
261  the existent value. It a iteration is running, and the item was already iterated, then
262  it will be updated on the `niterated_items` dict.
263  """
264  items_dictionary = self.items_dictionary
265 
266  if key in items_dictionary:
267  item_index = items_dictionary[key]
268  self.values_list[item_index] = value
269 
270  else:
271  empty_slots = self.empty_slots
272  values_list = self.values_list
273 
274  if empty_slots and self.new_items_skip_count > 0:
275  free_slot = empty_slots.pop()
276  self.filled_slots.add( free_slot )
277 
278  values_list[free_slot] = value
279  self.keys_list[free_slot] = key
280 
281  else:
282  free_slot = len( values_list )
283  values_list.append( value )
284  self.keys_list.append( key )
285 
286  self.items_dictionary[key] = free_slot

307  def trim_indexes_sorted(self):
308  """
309  Fix the the outdated indexes on the internal lists after their dictionary removal,
310  keeping the items original ordering O(n).
311  """
312  new_index = -1
313  clean_keys = []
314  clean_values = []

329  def trim_indexes_unsorted(self):
330  """
331  Fix the the outdated indexes on the internal lists after their dictionary removal.
332  keeping the items original ordering O(k), where `k` is length of `self.empty_slots`.
333  """
334  keys_list = self.keys_list
335  values_list = self.values_list
336  empty_slots = self.empty_slots
337  last_slot = -1
338 
339  while empty_slots:
340  empty_slot = empty_slots.pop()
341  list_length = len( keys_list )
342  key = keys_list.pop()
343 
344  keys_list[empty_slot] = key
345  values_list[empty_slot] = values_list.pop()
346  self.items_dictionary[key] = empty_slot
347 
348  if last_slot >= list_length:
349  last_slot = last_slot
350 
351  if last_slot > -1:
352  del keys_list[last_slot:]
353  del values_list[last_slot:]

397  def get_iterator(self, target_generation, how_many_times=-1):
398  """
399  Get fully configured iterable given the getter `target_generation(index)` function.
400 
401  `how_many_times` is for how many iterations it should keep ignoring the new items.
402  """
403  self.new_items_skip_count -= 1
404 
405  # When the current sequence is exhausted, search for an old one
406  if not self.new_items_skip_count > 0:
407 
408  # Unwind the stack until a valid item is found
409  if self.new_items_skip_stack:
410  self.new_items_skip_count = self.new_items_skip_stack.pop()
411 
412  self.not_iterate_over_new_items( how_many_times )
413 
414  if self.new_items_skip_count > 0:
415  return DynamicIterable( self.items_dictionary, target_generation, self.empty_slots, self.maximum_iterable_index, self.filled_slots )
416 
417  return DynamicIterable( self.items_dictionary, target_generation, self.empty_slots )
